public class negative {
    public static void main(String[] args) {
        int[] numbers = {16, -18, 27, -16, 23, -21, 19};
        int count = 0;

        for (int num : numbers) {
            if (num < 0) {
                count++;
            }
        }

        System.out.println("Number of negative numbers in the list: " + count);
    }
}
